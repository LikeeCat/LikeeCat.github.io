21. Merge Two Sorted Lists

2021-01-10
leetcode

题目

一些思考

  1. 首先这是两个分别排好序的数组
  2. 我们需要去遍历这两个链表,比较两个值的大小,然后插入新的链表中
  3. 注意一些可选值的判断

代码

class Solution {
 func mergeTwoLists(_ l1: ListNode?, _ l2: ListNode?) -> ListNode? {
    let headerNode = ListNode
        .init();
    var headerPoint = headerNode;
    var l1Pointer = l1
    var l2Pointer = l2
    while l1Pointer != nil || l2Pointer != nil {
        if l1Pointer != nil && l2Pointer != nil{
            if l1Pointer!.val >= l2Pointer!.val {
                headerPoint.next = l2Pointer
                l2Pointer = l2Pointer?.next
            }
            else{
                headerPoint.next = l1Pointer
                l1Pointer = l1Pointer?.next
            }
        }
        else{
            if l1Pointer == nil {
                headerPoint.next = l2Pointer
                l2Pointer = l2Pointer?.next
            }
            else{
                headerPoint.next = l1Pointer
                l1Pointer = l1Pointer?.next
            }
        }

        headerPoint = headerPoint.next!;

    }

    return headerNode.next;

}
}
